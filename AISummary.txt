COMMUNICATION_ARCHITECTURE:
  dual_layer:
    internal_bus:
      type: InProcessMessageBus (System.Threading.Channels)
      purpose: Real-time agent-to-agent signaling within process
      examples: "task-assigned", "review-requested", "tests-passed"
      latency: <1ms
      durability: none (in-memory only)
    github_layer:
      type: Octokit REST API
      purpose: Durable artifacts, human oversight, external coordination
      examples: PRs with code, Issues for tasks, Comments for discussion
      latency: 100-500ms
      durability: permanent (Git history)
    relationship: Complementary. Bus=instant internal signals. GitHub=durable artifacts+human oversight. Agent creates PR on GitHub AND sends bus message to notify PM.
